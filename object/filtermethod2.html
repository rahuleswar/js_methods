Using Filter Method 2

const students = [
{ name: 'Quincy', grade: 96 },
{ name: 'Jason', grade: 84 },
{ name: 'Alexis', grade: 100 },
{ name: 'Sam', grade: 65 },
{ name: 'Katie', grade: 90 }
];
var output = students.filter(function(value,index,array){
return value.grade > 87
});
document.write(JSON.stringify(output));
