Using Find Method

const employes = [
{ name: "David Carlson", age: 30 },
{ name: "John Cena", age: 34 },
{ name: "Mike Sheridan", age: 25 },
{ name: "John Carte", age: 50 }
];
var find = employes.find(function (name,index,arr) {
return name.age > 30
});
var findindex = employes.findIndex(function(name){
return name.age > 35
});
document.write(JSON.stringify(find));
document.write(JSON.stringify(findindex));
