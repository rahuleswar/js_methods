JS to display n terms of natural numbers and their sum

let n=parseInt(prompt("Enter the n numbers :"));
document.write("Enterd number is : ",n," ");
document.write(" ");
var sum = 0;
document.write("The first ",n," natural number is : ");
for (var i = 1; i <= n; i++) {
document.write(i," ");
sum += i;
}
document.write(" ");
document.write("The Sum of Natural Number upto ",n," terms : ",sum);
