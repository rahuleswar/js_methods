JS program to display the n terms of odd natural numbers and their sum

let n=parseInt(prompt("Enter the n numbers :"));
document.write("Enterd number is : ",n," ");
document.write(" ");
var sum = 0;
document.write("The odd numbers are : ");
for (var i = 1; i <= n; i++) {
document.write(2*i-1," ");
sum += 2*i-1;
}
document.write(" ");
document.write("Sum of Odd number : ",sum);
