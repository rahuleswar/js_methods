JS program to display the sum of n terms of even natural numbers

let n=parseInt(prompt("Enter the n numbers :"));
document.write("Enterd number is : ",n," ");
document.write(" ");
var sum = 0;
document.write("The n even numbers are : ");
for (var i = 1; i <= n; i++) {
document.write(2*i," ");
sum += 2*i;
}
document.write(" ");
document.write("Sum of even number : ",sum);
