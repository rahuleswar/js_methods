JS to read 10 numbers from the keyboard and find their sum and average

let n=parseInt(prompt("Enter the N numbers :"));
document.write("Enterd the N number is: ",n," ");
document.write(" ");
var sum = 0;
for (var i = 1; i <= n; i++) {
sum += i;
avg = sum/n;
}
document.write("The sum of 10 no is : ",sum," ");
document.write("The Average is : ",avg);
