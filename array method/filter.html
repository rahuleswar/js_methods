This is a competition. In this competition, there are three winners.(use filter method)

var winners = ["Anna","Beth","Cara"];
var result = winners.filter(function (name,index,arr) {
if (index+1 < 2) {
document.write(`Gold winner: ${name}. `);
}
else if(index+1 < arr.length){
document.write(`Silver Winner: ${name}. `);
}
else{
document.write(`Bronze Winner: ${name}.`);
}
});
document.write(result);
