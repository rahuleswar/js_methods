calculate the average of n elements in an array

let n=parseInt(prompt("Enter the N number :"));
var array = [];
var sum = 0;
for (var i = 0; i <= n; i++) {
array.push(i)
console.log(array)
sum += array[i]
console.log("total value : ",sum)
var count = array.length;
console.log("total subject : ",count)
var avrage = sum/count;
console.log("avrage of subject : ",avrage)
}

document.write("N number of arrays : ",array," ");
document.write("Avrage of given n numbers : ",avrage," ");
